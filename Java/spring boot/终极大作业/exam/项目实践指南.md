### 2. 导入数据库
用 Navicat 或 MySQL 命令行执行：
```sql
CREATE DATABASE smbms DEFAULT CHARACTER SET utf8mb4;
USE smbms;
-- 然后导入 smbms_db.sql（位于 第10周 综合项目1登录功能/smbms_db.sql）
```

数据库包含 5 张表：
| 表名 | 说明 | 数据量 |
|------|------|--------|
| `smbms_user` | 用户表 | 15条 |
| `smbms_role` | 角色表 | 3条（系统管理员/经理/普通用户） |
| `smbms_bill` | 订单表 | 18条 |
| `smbms_provider` | 供应商表 | 15条 |
| `smbms_address` | 地址表 | 6条 |

### 3. 确认项目能启动
在 IDEA 中直接运行 `SpringBootDemoApplication.main()`，访问 `http://localhost:8080/login` 应该能看到登录页面（虽然还没有后端逻辑）。

package com.example.springbootdemo.impl;

import com.example.springbootdemo.dao.UserMapper;
import com.example.springbootdemo.domain.Users;
import com.example.springbootdemo.service.UserService;
import org.springframework.beans.factory.annotation.Autowired;
import org.springframework.stereotype.Service;

@Service
public class UserServiceImpl implements UserService {

    @Autowired
    private UserMapper userMapper;

    @Override
    public Users login(String userCode, String userPassword) {
        Users user = userMapper.getUserByUserCode(userCode);
        if (user != null && user.getUserPassword().equals(userPassword)) {
            return user;
        }
        return null;
    }
}

### Step 5：创建 Controller

需要新创建包 `controller`，在里面创建：

**`src/main/java/com/example/springbootdemo/controller/MyController.java`**
```java
package com.example.springbootdemo.controller;

import com.example.springbootdemo.domain.Users;
import com.example.springbootdemo.service.UserService;
import org.springframework.beans.factory.annotation.Autowired;
import org.springframework.stereotype.Controller;
import org.springframework.ui.Model;
import org.springframework.web.bind.annotation.GetMapping;
import org.springframework.web.bind.annotation.PostMapping;
import org.springframework.web.bind.annotation.RequestParam;

import javax.servlet.http.HttpSession;

@Controller
public class MyController {

    @Autowired
    private UserService userService;

    // 访问 / 或 /login → 显示登录页
    @GetMapping({"/", "/login"})
    public String login() {
        return "login";
    }

    // 登录表单提交
    @PostMapping("/dologin")
    public String doLogin(@RequestParam String userCode,
                          @RequestParam String userPassword,
                          HttpSession session,
                          Model model) {
        Users user = userService.login(userCode, userPassword);
        if (user != null) {
            session.setAttribute("user", user);
            return "redirect:/welcome";
        } else {
            model.addAttribute("error", "用户名或密码错误！");
            return "login";
        }
    }

    // 欢迎页（需要登录才能访问）
    @GetMapping("/welcome")
    public String welcome(HttpSession session) {
        if (session.getAttribute("user") == null) {
            return "redirect:/login";
        }
        return "welcome";
    }
}
```

### Step 3：更新 UserMapper.xml（增加分页 + 关联查询 SQL）

在 `src/main/resources/mapper/UserMapper.xml` 中，**保留原有 `getUserByUserCode`，在 `</mapper>` 前追加**：

```xml
<!-- 分页查询用户列表（多表联查 + 条件搜索） -->
<select id="getUserListByPage" resultMap="UserWithRoleName">
    select u.*,r.* from smbms_user u join smbms_role r on u.userRole = r.id
    <where>
        <if test="userName != null and userName != ''">
            and u.userName like CONCAT ('%',#{userName},'%')
        </if>
        <if test="userRole != null and userRole != 0">
            and u.userRole = #{userRole}
        </if>
    </where>
    order by u.id
</select>

<!-- 结果映射：关联角色信息 -->
<resultMap id="UserWithRoleName" type="Users">
    <id property="id" column="id" />
    <association property="role" javaType="SmbmsRole">
        <id property="id" column="id" />
    </association>
</resultMap>
```
