建议下面这部分内容默认挂载到WSL中，因为在windows下会有启动依赖，容器启动后不能迅速访问(要等)
如果实在要把这个数据挂载要Windows下就得创建data文件夹和`kvrocks.conf`在这个挂载的目录下，并且还要写入内容在`kvrocks.conf`中
```yaml
  cvat_redis_ondisk:
    volumes:
      - ../CVAT_DATA/cvat_cache_db:/var/lib/kvrocks
      # 你需要确保 kvrocks.conf 文件存在于改目录下（写入内容）
      # bind 0.0.0.0
      # port 6666
      # daemonize no
      # logfile stdout
      # dir /var/lib/kvrocks/data
```
