# 三、集合操作

### 1.创建基础集合`baseCollection`

```
db.createCollection('baseCollection')
```

### 2.创建固定大小集合`cappedCollection`

```
db.createCollection(
	'cappedCollection',
	{ 
		capped : true, 
		size : 5242880, 
		max : 5000 
	}
)
```

### 3.创建时间序列集合`weather24h`

```
db.createCollection(
    'weather24h',
    {
       timeseries: {
          timeField: 'timestamp',
          metaField: 'data',
          granularity: 'hours'
       },
       expireAfterSeconds: 86400
    }
)
```

### 4.创建集群化集合`clusteredCollection`

```
db.createCollection(
    'clusteredCollection',
    { 
		clusteredIndex: {
			key: { _id: 1 }, 
			unique: true, 
			name: 'clusteredKey' 
		} 
	}
)
```

### 5.创建视图`myView`

```
db.createCollection(
  'myView',
  {
    viewOn: 'baseCollection',
    pipeline: [{ $match: { year: 1 } }]
  }
)
```

### 6.查看集合`baseCollection`的大小

```
db.baseCollection.dataSize()
```

### 7.查看集合`cappedCollection`的信息

```
db.getCollectionInfos({name: 'cappedCollection'})
```

### 8.查看集合`cappedCollection` 的文档计数  

```
db.cappedCollection.countDocuments()
```

### 9.查看当前数据库中的所有集合

```
show collections
```

### 10.删除集合`clusteredCollection`

```
db.clusteredCollection.drop()
```

### 11.修改集合`baseCollection`的名称

```
db.baseCollection.renameCollection('baseCollectionNew')
```

# 五、文档操作：查询文档

### 1.基本查询

```
db.users.find()
db.users.find({},{username: 1,'address.city': 1})
```

### 2.条件查询

```
db.users.find({age: {$gt: 29}},{age: 1})
db.users.find({age: {$lt: 29}},{age: 1})
db.users.find({age: {$gte: 28}},{age: 1})
db.users.find({age: {$lte: 28}},{age: 1})
db.users.find({signature: {$eq: null}})
db.users.find({'address.province': {$in: ['四川省','云南省']}},{address: 1})
db.users.find({'is_active': {$ne: true}},{is_active: 1})
db.users.find(
	{'address.province': {$nin: ['四川省','云南省']}},
	{address: 1}
)
db.users.find({$and: [{age: {$gt: 25}},{age: {$lt: 30}}]},{age: 1})
db.users.find({$or: [{age: {$lt: 28}},{age: {$gt: 30}}]},{age: 1})
db.users.find({signature: {$regex: /心/}},{signature: 1})
db.users.find({interests: {$all: ['旅行','读书']}})
db.users.find({interests: {$size: 2}},{interests: 1})
db.users.find({_id: {$type: 'int'}})
db.users.find({badges: {$exists: true}})
```

### 3.去重查询

```
db.users.distinct('age')
```

### 4.排序查询

```
db.users.find({},{_id: 0,username: 1,age: 1}).sort({age: -1})
db.users.find({},{_id: 0,username: 1,age: 1}).sort({age: -1}).limit(3)
```

# 六、文档操作：更新文档

### 1.更新单个文档

```
//更新当个文档
db.users.updateOne(
    {_id: 1},
    {
        $set: {
            username: '追梦少年心',
            'address.district': '城阳区',
            interests: ['读书','旅行','音乐'],
            badges: '创作达人'
        },
        $currentDate: {lastModified: true}
    }
)
db.users.find({_id: {$eq: 1}})
//替换单个文档
db.users.replaceOne({username: '运动爱好者'},{username: null})
db.users.find({username: null})
```

### 2.更新多个文档

```
db.users.updateMany(
	{is_active: {$eq: false}},
	{
        $set: {
            is_active: true
        },
        $currentDate: {lastModified: true}
    }
)
db.users.find({is_active: {$eq: false}})
```

### 2.管道操作符$group

```
db.orders.aggregate(
	{
	  $group: 
		{
		  _id: '$productType',
		  totalQuantity: {$sum: '$salesQuantity'},
		  avgQuantity: {$avg: '$salesQuantity'},
		  maxQuantity: {$max: '$salesQuantity'},
		  minQuantity: {$min: '$salesQuantity'},
		  numCount: {$count: {}},
		  saleInfoTop2: {
                $topN: {
                n: 2,
                sortBy: {salesQuantity: -1},
                output: ['$_id','$productName','$salesQuantity']
                }
		    }
		}
	}
)
```

### 3.管道操作符$sort

```
db.orders.aggregate({$sort: {productType: 1,salesQuantity: -1}})
```

### 4.管道操作符$limit

```
db.orders.aggregate(
	[
	{$sort: {salesQuantity: -1}},
	{$limit: 3}
	]
)
```

### 5.管道操作符$project

```
db.orders.aggregate(
	{$project: 
	  {_id: 0,
	  totalSale: {$multiply: ['$unitPrice','$salesQuantity']},
	  cmpTo100: {$cmp: ['$salesQuantity',100]},
	  year: {$year: {
	  			date: '$orderDate',
	  			timezone: 'Asia/Shanghai'
	  		}
	  },
	  diffDate: {$dateDiff: {
	  			startDate: '$orderDate',
	  			endDate: new Date(),
	  			unit: 'day',
	  			timezone: 'Asia/Shanghai'
	  		}
	  },
	  stringDate: {$dateToString: {
	  			date: '$orderDate',
	  			format: '%Y-%m-%d %H:%M:%S',
	  			timezone: 'Asia/Shanghai'
	  		}
	  },
	  nameAndType: {$concat: ['$productName','-','$productType']}
	}
  }
)
```

### 6.管道操作符$sortByCount

```
db.orders.aggregate({$sortByCount: '$productType'})
```

### 7.管道操作符$merge

```
db.orders.aggregate(
	[
	  {$match: {productName: {$eq: '牛仔裤'}}},
	  {$merge: 'ordersNew'}
	]
)

db.ordersNew.find()

db.orders.aggregate(
	[
	  {$match: {productName: {$eq: '牛仔裤'}}},
	  {$project: {totalSale: 
	               {$multiply: 
	                 ['$unitPrice','$salesQuantity']
	               }
	             }
	  },
	  {$merge: 'ordersNew'}
	]
)
```

### 8.管道操作符$unionWith

```
db.orders.aggregate(
    [
        {$match: {productType: {$eq: '服装'}}},
        {$unionWith: {
        	coll: 'ordersNew',
        	pipeline: [{$project: {_id: 0,productName: 1,totalSale: 1}}]
        	}
        }
    ]
)
```

# 十一、安全操作：用户管理

```

-u 'itcastAdmin' -p

use admin

db.auth('itcastAdmin','123456')

show collections2
```

### 2.创建用户

```
use admin

db.auth('itcastAdmin','123456')

use my_database

db.createUser({
    user: 'myUser',
    pwd: '123456',
    roles: ['read',{role: 'read',db: 'test'}]
})
```

### 3.查看用户

```
db.getUser('myUser')
```

### 4.修改用户

```
db.updateUser(
	'myUser',
	{roles :['readWrite',{role: 'readWrite',db: 'test'}]}
)

db.grantRolesToUser('myUser',['dbAdmin'])
```

### 4.删除用户

```
db.dropUser('myUser')

db.getUsers()
```
